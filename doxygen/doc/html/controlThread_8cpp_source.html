22 void ControlThread::apply_ratio_limiter (double& linear_speed, double& angular_speed)
23 {
24  if (lin_ang_ratio>1.0) lin_ang_ratio = 1.0;
25  if (linear_speed>100) linear_speed = 100;
26  if (linear_speed<-100) linear_speed = -100;
27  if (angular_speed>100) angular_speed = 100;
28  if (angular_speed<-100) angular_speed = -100;
29 
30  double tot = fabs(linear_speed) + fabs(angular_speed);
31 
32  if (tot> 100)
33  {
34  //if lin_ang_ratio is negative, coeff will be 0 and
35  //the adaptive limiter will be used (current ratio)
36  double coeff = 0.0;
37  if (lin_ang_ratio>0.0) coeff = (tot-100.0)/100.0;
38  double angular_speed_A, angular_speed_B, linear_speed_A, linear_speed_B;
39 
40  angular_speed_A = angular_speed *(1-lin_ang_ratio);
41  linear_speed_A = linear_speed * lin_ang_ratio;
42 
43  double current_ratio = fabs(linear_speed/angular_speed);
44  if (angular_speed>0) angular_speed_B = 100.0/(current_ratio+1.0);
45  else angular_speed_B = -100.0/(current_ratio+1.0);
46 
47  linear_speed_B = 100.0-fabs(angular_speed_B);
48  //if (angular_speed>0) linear_speed_B = 100.0-fabs(angular_speed_B);
49  //else linear_speed_B = -100.0+fabs(angular_speed_B);
50 
51  linear_speed = linear_speed_A * (coeff) + linear_speed_B * (1.0-coeff);
52  angular_speed = angular_speed_A * (coeff) + angular_speed_B * (1.0-coeff);
53 
54  }
55 }

57 void ControlThread::apply_ratio_limiter (double max, double& linear_speed, double& angular_speed)
58 {
59  if (lin_ang_ratio<0.0) lin_ang_ratio = 0.0;
60  if (lin_ang_ratio>1.0) lin_ang_ratio = 1.0;
61  if (linear_speed > max*lin_ang_ratio) linear_speed = max*lin_ang_ratio;
62  if (linear_speed < -max*lin_ang_ratio) linear_speed = -max*lin_ang_ratio;
63  if (angular_speed > max*(1-lin_ang_ratio)) angular_speed = max*(1-lin_ang_ratio);
64  if (angular_speed < -max*(1-lin_ang_ratio)) angular_speed = -max*(1-lin_ang_ratio);
65 }

229 void ControlThread::run()
230 {
231  this->odometry_handler->compute();
232 
233  double pidout_linear_speed = 0;
234  double pidout_angular_speed = 0;
235  double pidout_direction = 0;
236 
237  //input_linear_speed and input_angular speed ranges are: 0-100
238  this->motor_handler->read_inputs(&input_linear_speed, &input_angular_speed, &input_desired_direction, &input_pwm_gain);
239  apply_input_filter(input_linear_speed, input_angular_speed,input_desired_direction);
240  apply_ratio_limiter(input_linear_speed, input_angular_speed);
241 
242  if (!lateral_movement_enabled)
243  {
244  if (input_desired_direction>-90 && input_desired_direction <90) input_desired_direction = 0;
245  else if (input_desired_direction <= -90) input_desired_direction = 180;
246  else if (input_desired_direction >= +90) input_desired_direction = 180;
247  }
248 
249  exec_pwm_gain = input_pwm_gain / 100.0 * 1.0;
250  exec_desired_direction = input_desired_direction;
251 
252  //The controllers
253  double MAX_VALUE = 0;
254  if (ikart_control_type == IKART_CONTROL_OPENLOOP_NO_PID)
255  {
256  MAX_VALUE = 1250; // Maximum joint PWM
257  exec_linear_speed = input_linear_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
258  exec_angular_speed = input_angular_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
259 
260  pidout_linear_speed = exec_linear_speed;
261  pidout_angular_speed = exec_angular_speed;
262  pidout_direction = exec_desired_direction;
263  this->motor_handler->execute_openloop(pidout_linear_speed,pidout_direction,pidout_angular_speed);
264  }
265  else if (ikart_control_type == IKART_CONTROL_VELOCITY_NO_PID)
266  {
267  MAX_VALUE = 200; // Maximum joint speed (deg/s)
268  exec_linear_speed = input_linear_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
269  exec_angular_speed = input_angular_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
270 
271 //#define PRINT_CURRENT_VEL
272 #ifdef PRINT_CURRENT_VEL
273  //printf("%+5.5f, %+5.5f, %+5.5f\n", input_angular_speed /100 * this->motor_handler->get_max_angular_vel(), this->odometry_handler->ikart_vel_theta, exec_angular_speed);
274  printf("%+5.5f, %+5.5f, %+5.5f\n", input_linear_speed /100 * this->motor_handler->get_max_linear_vel(), this->odometry_handler->ikart_vel_lin, exec_linear_speed);
275 #endif
276 
277  pidout_linear_speed = exec_linear_speed;
278  pidout_angular_speed = exec_angular_speed;
279  pidout_direction = exec_desired_direction;
280  this->motor_handler->execute_speed(pidout_linear_speed,pidout_direction,pidout_angular_speed);
281  }
282  else if (ikart_control_type == IKART_CONTROL_OPENLOOP_PID)
283  {
284  MAX_VALUE = 1250; // Maximum joint PWM
285  exec_linear_speed = input_linear_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
286  exec_angular_speed = input_angular_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
287 
288  apply_control_openloop_pid(pidout_linear_speed,pidout_angular_speed,pidout_direction,exec_linear_speed,exec_angular_speed,exec_desired_direction);
289  this->motor_handler->execute_speed(pidout_linear_speed,pidout_direction,pidout_angular_speed);
290  }
291  else if (ikart_control_type == IKART_CONTROL_VELOCITY_PID)
292  {
293  MAX_VALUE = 200; // Maximum joint speed (deg/s)
294  exec_linear_speed = input_linear_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
295  exec_angular_speed = input_angular_speed / 100.0 * MAX_VALUE * exec_pwm_gain;
296 
297  apply_control_speed_pid(pidout_linear_speed,pidout_angular_speed,pidout_direction,exec_linear_speed,exec_angular_speed,exec_desired_direction);
298 
299  pidout_angular_speed += exec_angular_speed;
300  this->motor_handler->execute_speed(pidout_linear_speed,pidout_direction,pidout_angular_speed);
301  }
302  else
303  {
304  yError ("Unknown control mode!");
305  exec_linear_speed = 0;
306  exec_angular_speed = 0;
307  exec_pwm_gain = 0;
308  exec_desired_direction = 0;
309  this->motor_handler->execute_none();
310  }
311 }

320 bool ControlThread::set_control_type (string s)
321 {
322  if (s == "none") ikart_control_type = IKART_CONTROL_NONE;
323  else if (s == "velocity_no_pid") ikart_control_type = IKART_CONTROL_VELOCITY_NO_PID;
324  else if (s == "openloop_no_pid") ikart_control_type = IKART_CONTROL_OPENLOOP_NO_PID;
325  else if (s == "velocity_pid") ikart_control_type = IKART_CONTROL_VELOCITY_PID;
326  else if (s == "openloop_pid") ikart_control_type = IKART_CONTROL_OPENLOOP_PID;
327  else
328  {
329  yError("Error: unknown type of control required: %s. Closing...\n",s.c_str());
330  return false;
331  }
332  yInfo("Control type set to: %s\n",s.c_str());
333  return true;
334 }

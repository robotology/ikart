148  double lp_filter_1Hz(double input, int i)
149  {
150  //This is a butterworth low pass first order, with a cut off freqency of 1Hz
151  //It must be used with a sampling frequency of 50Hz (20ms)
152  static double xv[2][10], yv[2][10];
153  xv[0][i] = xv[1][i];
154  xv[1][i] = input /1.689454484e+01;
155  yv[0][i] = yv[1][i];
156  yv[1][i] = (xv[0][i] + xv[1][i]) + ( 0.8816185924 * yv[0][i]);
157  return yv[1][i];
158  }
159  double lp_filter_0_5Hz(double input, int i)
160  {
161  //This is a butterworth low pass first order, with a cut off freqency of 0.5Hz
162  //It must be used with a sampling frequency of 50Hz (20ms)
163  static double xv[2][10], yv[2][10];
164  xv[0][i] = xv[1][i];
165  xv[1][i] = input /3.282051595e+01;
166  yv[0][i] = yv[1][i];
167  yv[1][i] = (xv[0][i] + xv[1][i]) + ( 0.9390625058 * yv[0][i]);
168  return yv[1][i];
169  }

171  virtual void run()
172  {
173  yarp::sig::Vector* l_wrench = l_wrench_in_port.read(false);
174  yarp::sig::Vector* r_wrench = r_wrench_in_port.read(false);
175 
176  if (l_wrench!=0)
177  {
178  ly=-(l_wrench->data()[0]-lx0);
179  lx=+(l_wrench->data()[1]-ly0);
180  }
181 
182  if (r_wrench!=0)
183  {
184  ry=-(r_wrench->data()[0]-rx0);
185  rx=+(r_wrench->data()[1]-ry0);
186  }
187 
188  lx = lp_filter_0_5Hz(lx,0);
189  ly = lp_filter_0_5Hz(ly,1);
190  rx = lp_filter_0_5Hz(rx,2);
191  ry = lp_filter_0_5Hz(ry,3);
192 
193  double linear_gain = 60; // 60N = 100%
194  double angular_gain = 60; // 60Nm = 100%
195  double desired_direction = atan2( lx+rx, ly+ry ) * 180.0 / 3.14159265;
196  double linear_speed = sqrt ( pow(lx+rx,2)+ pow(ly+ry,2) ) / linear_gain * 100;
197  double angular_speed = (ly-ry) / angular_gain * 100;
198  const double pwm_gain = 100;
199 
200  //dead band
201  if (fabs(angular_speed)<10) angular_speed = 0;
202  if (fabs(linear_speed) <10) linear_speed = 0;
203 
204  //saturation
205  linear_speed= (linear_speed > +100) ? +100:linear_speed;
206  linear_speed= (linear_speed < -100) ? -100:linear_speed;
207  angular_speed= (angular_speed > +100) ? +100:angular_speed;
208  angular_speed= (angular_speed < -100) ? -100:angular_speed;
209 
210  printf ("(%+8.2f %+8.2f)(%+8.2f %+8.2f) %+9.1f %+9.1f %+9.1f %+8.0f\n",lx,ly,rx,ry,desired_direction,linear_speed,angular_speed,pwm_gain);
211 
212 
213  //send data to yarp output port
214  Bottle bot;
215  bot.addInt(1);
216  bot.addDouble(desired_direction);
217  bot.addDouble(linear_speed);
218  bot.addDouble(angular_speed);
219  bot.addDouble(pwm_gain);
220  commands_out_port.write(bot);
221 
222  }

119 /* Expose callback for the drawing area */
120 bool GraphicsManager::on_drawingarea_expose(GdkEventExpose *event)
121 {
122  gint rowstride = m_refPixbuf->get_rowstride();
123 
124  const guchar* pixels = m_refPixbuf->get_pixels() + (rowstride * event->area.y) + (event->area.x * 3);
125 
126  Glib::RefPtr<Gdk::Window> refWindow = m_DrawingArea.get_window();
127  Glib::RefPtr<Gdk::GC> refGC = m_DrawingArea.get_style()->get_black_gc();
128 
129  refWindow->draw_rgb_image_dithalign(refGC,
130  event->area.x, event->area.y,
131  event->area.width, event->area.height,
132  Gdk::RGB_DITHER_NORMAL,
133  pixels, rowstride,
134  event->area.x, event->area.y);
135 
136  return true;
137 }

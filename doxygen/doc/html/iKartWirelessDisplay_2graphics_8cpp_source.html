71 void GraphicsManager::load_pixbufs()
72 {
73  if(m_refPixbuf_Background)
74  return; /* already loaded earlier */
75 
76  Glib::RefPtr<Gdk::Pixbuf> tmp_Pixbuf_Numbers;
77  Glib::RefPtr<Gdk::Pixbuf> tmp_Pixbuf_Blocks;
78  Glib::RefPtr<Gdk::Pixbuf> tmp_Pixbuf_Charge;
79 
80  printf ("loading: %s\n", picBackgroud.c_str());
81  try
82  {
83  m_refPixbuf_Background = Gdk::Pixbuf::create_from_file(picBackgroud.c_str());
84  }
85  catch (Glib::FileError e)
86  {
87  g_message("caught Glib::FileError %d", e.code());
88  return;
89  }
90  catch (Gdk::PixbufError e)
91  {
92  g_message("Gdk::PixbufError: %d",e.code());
93  return;
94  }
95 
96  printf ("loading: %s\n", picNumbers.c_str());
97  tmp_Pixbuf_Numbers = Gdk::Pixbuf::create_from_file(picNumbers.c_str());
98 
99  printf ("loading: %s\n", picBlocks.c_str());
100  tmp_Pixbuf_Blocks = Gdk::Pixbuf::create_from_file(picBlocks.c_str());
101 
102  m_refPixbuf_Numbers = tmp_Pixbuf_Numbers->add_alpha (true, 255,0,255);
103  m_refPixbuf_Blocks = tmp_Pixbuf_Blocks->add_alpha (true, 255,0,255);
104 
105  m_back_width = m_refPixbuf_Background->get_width();
106  m_back_height = m_refPixbuf_Background->get_height();
107 }

111 bool GraphicsManager::on_drawingarea_expose(GdkEventExpose *event)
112 {
113  gint rowstride = m_refPixbuf->get_rowstride();
114 
115  const guchar* pixels = m_refPixbuf->get_pixels() + (rowstride * event->area.y) + (event->area.x * 3);
116 
117  Glib::RefPtr<Gdk::Window> refWindow = m_DrawingArea.get_window();
118  Glib::RefPtr<Gdk::GC> refGC = m_DrawingArea.get_style()->get_black_gc();
119 
120  refWindow->draw_rgb_image_dithalign(refGC,
121  event->area.x, event->area.y,
122  event->area.width, event->area.height,
123  Gdk::RGB_DITHER_NORMAL,
124  pixels, rowstride,
125  event->area.x, event->area.y);
126 
127  return true;
128 }

136  //signal = 100; //for debugging

144  int off;

159  ypos=26;

165  }
166  }

168  sprintf(buff,"%4.1f",fabs(strenght));
169  len = strlen(buff);
170  point_off=0;
171  for (int i=0;i<len;i++)
172  {
173  int xoff=0;
174  int yoff=0;
175  int xpos=0;
176  int ypos=0;
177  GdkRectangle dest;
178  if (buff[i]=='.')
179  point_off=17;
180 
181  if (buff[i]>='0' && buff[i]<='9')
182  {
183  xoff=(buff[i]-'0')*29+point_off;
184  dest.x=63+i*29-point_off;

187  dest.height=52;

189  ypos=86-yoff;

196  }

198  //draw charge indicator

201  {

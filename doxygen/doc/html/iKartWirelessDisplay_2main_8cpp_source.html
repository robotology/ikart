112  virtual bool on_timeout()
113  {
114  char cmd[] = "lynx -auth=admin:password -dump \" http://10.0.0.250/cgi-bin/cgi?req=frm&frm=info.html&rand=1616217005\" | grep -i status";
115  FILE* pipe =popen(cmd,"r");
116  if (!pipe) return "ERROR";
117  char buffer[128];
118  string result = "";
119  while(!feof(pipe)) {
120  if(fgets(buffer, 128, pipe) != NULL)
121  result += buffer;
122  }
123  pclose(pipe);
124 
125  static int signal=0;
126  char tmp[20];
127  static double strenght=0.0;
128 
129  sscanf(result.c_str(),"%*s %*s %d %*s %s",&signal,tmp );
130  strenght = atof (tmp+1);
131 
132  Bottle& outBot = monitorOutput.prepare();
133  outBot.clear();
134  outBot.addInt(signal);
135  outBot.addDouble(strenght);
136  monitorOutput.write();
137 
138  graphics->update_graphics(signal,strenght);
139 
140  return true;
141  }

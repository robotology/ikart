43 int GotoThread::pnpoly(int nvert, double *vertx, double *verty, double testx, double testy)
44 {
45  int i, j, c = 0;
46  for (i = 0, j = nvert-1; i < nvert; j = i++) {
47  if ( ((verty[i]>testy) != (verty[j]>testy)) &&
48  (testx < (vertx[j]-vertx[i]) * (testy-verty[i]) / (verty[j]-verty[i]) + vertx[i]) )
49  c = !c;
50  }
51  return c;
52 }

55 bool GotoThread::compute_obstacle_avoidance()
56 {
57  double min_distance = max_obstacle_distance;
58  double min_angle = 0.0;
59 
60  for (size_t i=0; i<1080; i++)
61  {
62  double curr_d = laser_data.get_distance(i);
63  double curr_angle = laser_data.get_angle(i);
64  size_t angle_t = (size_t)(4.0 * frontal_blind_angle);
65  if (i>=540-angle_t && i<=540+angle_t) continue; //skip frontalobstacles
66 
67  if (curr_d < min_distance)
68  {
69  min_distance = curr_d;
70  min_angle = curr_angle;
71  }
72  }
73 
74  angle_f = min_angle;
75  angle_t = angle_f+90.0;
76  w_f = (1-(min_distance/max_obstacle_distance))/2;
77  w_t = 0;
78  return true;
79 }

81 bool GotoThread::check_obstacles_in_path()
82 {
83  int laser_obstacles = 0;
84  double goal_distance = 1000; //TO BE COMPLETED
85 
86  //compute the polygon
87  double vertx[4];
88  double verty[4];
89  double theta = 0.0;
90  double ctheta = cos(theta);
91  double stheta = sin(theta);
92  double detection_distance = 1.5;
93  if(enable_dynamic_max_distance)
94  detection_distance = max_detection_distance * safety_coeff;
95  else
96  detection_distance = max_detection_distance;
97  if (detection_distance<min_detection_distance) detection_distance=min_detection_distance;
98  vertx[0]=(-robot_radius) * ctheta + detection_distance * (-stheta);
99  verty[0]=(-robot_radius) * stheta + detection_distance * ctheta;
100  vertx[1]=(+robot_radius) * ctheta + detection_distance * (-stheta);
101  verty[1]=(+robot_radius) * stheta + detection_distance * ctheta;
102  vertx[2]= +robot_radius * ctheta;
103  verty[2]= +robot_radius * stheta;
104  vertx[3]= -robot_radius * ctheta;
105  verty[3]= -robot_radius * stheta;
106 
107  for (size_t i=0; i<1080; i++)
108  {
109  double d = laser_data.get_distance(i);
110  if (d < robot_radius)
111  {
112  laser_obstacles++;
113  printf("obstacles on the platform\n");
114  continue;
115  }
116 
117  double px= laser_data.get_x(i);
118  double py= laser_data.get_y(i);
119  if (pnpoly(4,vertx,verty,px,py)>0)
120  {
121  double d = laser_data.get_distance(i);
122  if (d < goal_distance)
123  //if (laser_data.get_distance(i) < goal_distance)
124  {
125  laser_obstacles++;
126  //printf("obstacles on the path\n");
127  continue;
128  }
129  else
130  {
131  //printf("obstacles on the path, but goal is near\n");
132  continue;
133  }
134  }
135  }
136 
137  //prevent noise to be detected as an obtacle;
138  if (laser_obstacles>=2)
139  {
140  printf("obstacles detected\n");
141  return true;
142  }
143 
144  //no obstacles found
145  return false;
146 }

148 void GotoThread::run()
149 {
150  mutex.wait();
151 
152  //data is formatted as follows: x, y, angle
153  yarp::sig::Vector *loc = port_localization_input.read(false);
154  if (loc) {localization_data = *loc; loc_timeout_counter=0;}
155  else
156  {
157  if (use_localization) loc_timeout_counter++;
158  if (loc_timeout_counter>TIMEOUT_MAX) loc_timeout_counter=TIMEOUT_MAX;
159  }
160  if (loc_timeout_counter>=TIMEOUT_MAX)
161  {
162  if (status == MOVING)
163  {
164  printf ("stopping navigation because of localization timeouts!");
165  status = ABORTED;
166  }
167  }
168 
169  yarp::sig::Vector *odm = port_odometry_input.read(false);
170  if (odm) {odometry_data = *odm; odm_timeout_counter=0;}
171  else
172  {
173  if (use_odometry) odm_timeout_counter++;
174  if (odm_timeout_counter>TIMEOUT_MAX) odm_timeout_counter=TIMEOUT_MAX;
175  }
176  if (odm_timeout_counter>=TIMEOUT_MAX)
177  {
178  if (status == MOVING)
179  {
180  printf ("stopping navigation because of odometry timeouts!");
181  status = ABORTED;
182  }
183  }
184 
185  yarp::os::Bottle *scan = port_laser_input.read(false);
186  if (scan) {laser_data.set_cartesian_laser_data(scan); las_timeout_counter=0;}
187  else {las_timeout_counter++; if (las_timeout_counter>TIMEOUT_MAX) las_timeout_counter=TIMEOUT_MAX;}
188 
189  //computes the control action
190  control_out.zero();
191 
192  //gamma is the angle between the current ikart heading and the target heading
193  double unwrapped_localization_angle = (localization_data[2]<0)?localization_data[2]+360:localization_data[2];
194  double unwrapped_target_angle = (target_data[2]<0)?target_data[2]+360:target_data[2];
195  //double gamma = localization_data[2]-target_data[2];
196  double gamma = unwrapped_localization_angle-unwrapped_target_angle;
197  if (gamma > 180) gamma -= 360;
198  else if (gamma < -180) gamma += 360;
199 
200  //beta is the angle between the current ikart position and the target position
201  double old_beta = atan2 (localization_data[1]-target_data[1],localization_data[0]-target_data[0])*180.0/M_PI;
202  double beta = -atan2 (target_data[1]-localization_data[1],target_data[0]-localization_data[0])*180.0/M_PI;
203 
204  //distance is the distance between the current ikart position and the target position
205  double distance = sqrt(pow(target_data[0]-localization_data[0],2) + pow(target_data[1]-localization_data[1],2));
206 
207  //compute the control law
208 // control_out[0] = -beta;
209 // control_out[0] = -beta-localization_data[2]; //CHECKME
210 // control_out[0] = -(beta-localization_data[2]); //CHECKME -180
211 // control_out[0] = -(beta+localization_data[2]); //CHECKME -90
212 // control_out[0] = +beta+localization_data[2]-90; //CHECKME -90
213 // control_out[0] = +beta-localization_data[2]-90; //CHECKME -90
214 // control_out[0] = -beta+localization_data[2]-90; //CHECKME -90
215 // control_out[0] = -beta-localization_data[2]-90; //CHECKME -90
216 // control_out[0] = -beta-localization_data[2]+90; //CHECKME -90
217 // control_out[0] = -beta+localization_data[2]+90; //CHECKME -90
218 // control_out[0] = +beta+localization_data[2]+90; //CHECKME -90
219 // control_out[0] = +beta-localization_data[2]+90; //CHECKME -90
220 // control_out[0] = -beta+localization_data[2]; //CHECKME -90
221 // control_out[0] = beta-localization_data[2]; //CHECKME -90
222  double tmp1= 180-(old_beta-localization_data[2]);
223  if (tmp1>360)
224  tmp1-=360;
225  if (tmp1>180 && tmp1<360)
226  tmp1 = tmp1-360;//ADDED LATER
227  //printf ("%f \n", control[0]);
228  control_out[1] = k_lin_gain * distance;
229  control_out[2] = k_ang_gain * gamma;
230  control_out[0] = tmp1;
231 
232  //control saturation
233  //printf ("%f %f ", control_out[2], control_out[1]);
234  if (control_out[2] > 0 )
235  {
236  if (control_out[2] > +max_ang_speed) control_out[2] = +max_ang_speed;
237  if (control_out[2] < +min_ang_speed) control_out[2] = +min_ang_speed;
238  }
239  else
240  {
241  if (control_out[2] < -max_ang_speed) control_out[2] = -max_ang_speed;
242  if (control_out[2] > -min_ang_speed) control_out[2] = -min_ang_speed;
243  }
244 
245  if (control_out[1] > 0 )
246  {
247  if (control_out[1] > +max_lin_speed) control_out[1] = +max_lin_speed;
248  if (control_out[1] < +min_lin_speed) control_out[1] = +min_lin_speed;
249  }
250  else
251  {
252  if (control_out[1] < -max_lin_speed) control_out[1] = -max_lin_speed;
253  if (control_out[1] > -min_lin_speed) control_out[1] = -min_lin_speed;
254  }
255  //printf ("%f %f \n", control_out[2], control_out[1]);
256 
257  //check for large rotations: inhibit linear movement, to allow a rotation on place
258  if (fabs(gamma)>25.0)
259  {
260  control_out[1] = 0;
261  }
262 
263  //check for obstacles, always performed
264  bool obstacles_in_path = false;
265  if (las_timeout_counter < 300)
266  {
267  obstacles_in_path = check_obstacles_in_path();
268  compute_obstacle_avoidance();
269  double correction = angle_f;
270  if (correction<0) correction+=180;
271  else correction-=180;
272  double w_f_sat = w_f;
273  if (w_f_sat>0.3) w_f_sat=0.3;
274  double goal = control_out[0];
275  double goal_corrected = goal * (1-w_f_sat) + correction * (w_f_sat);
276  if (enable_obstacles_avoidance)
277  {
278  //direction is modified in proximity of the obstacles
279  control_out[0] = goal_corrected;
280  //speed is reduced in proximity of the obstacles
281  double w_f_sat2 = w_f*2.2;
282  if (w_f_sat2>0.85) w_f_sat2= speed_reduction_factor; //CHECK 0.85, put it config file
283  control_out[1] = control_out[1] * (1.0-w_f_sat2);
284  }
285  else
286  {
287  control_out[0] = goal;
288  }
289  angle_g = goal_corrected;
290  }
291  else
292  {
293  if (status == MOVING)
294  {
295  }
296  }
297 
298  double current_time = yarp::os::Time::now();
299  double speed_ramp = (current_time-obstacle_removal_time)/2.0;
300 
301  switch (status.getStatusAsInt())
302  {
303  case MOVING:
304  //Update the safety coefficient only if your are MOVING.
305  //If you are WAITING_OBSTACLE, use the last valid safety_coeff until the
306  //obstacle has been removed.
307  safety_coeff = control_out[1]/max_lin_speed;
308 
309  //compute the speed ramp after the removal of an obstacle
310  speed_ramp = (speed_ramp > 1.0) ? 1.0 : speed_ramp;
311  control_out[1] *= speed_ramp;
312  control_out[2] *= speed_ramp;
313 
314  if (target_data.weak_angle)
315  {
316  //check if the goal has been reached in position but not in orientation
317  if (fabs(distance) < goal_tolerance_lin)
318  {
319  status=REACHED;
320  fprintf (stdout, "Goal reached!\n");
321  }
322  }
323  else
324  {
325  //check if the goal has been reached in both position and orientation
326  if (fabs(distance) < goal_tolerance_lin && fabs(gamma) < goal_tolerance_ang)
327  {
328  status=REACHED;
329  fprintf (stdout, "Goal reached!\n");
330  }
331 
332  }
333 
334  // check if you have to stop beacuse of an obstacle
335  if (enable_obstacles_emergency_stop && obstacles_in_path)
336  {
337  fprintf (stdout, "Obstacles detected, stopping \n");
338  status=WAITING_OBSTACLE;
339  obstacle_time = current_time;
340  Bottle b;
341  b.addString("Obstacles detected");
342  Bottle tmp = port_speak_output.prepare();
343  tmp.clear();
344  tmp=b;
345  port_speak_output.write();
346  }
347  break;
348 
349  case WAITING_OBSTACLE:
350  if (!obstacles_in_path)
351  {
352  if (fabs(current_time-obstacle_time)>1.0)
353  {
354  fprintf (stdout, "Obstacles removed, thank you \n");
355  status=MOVING;
356  Bottle b;
357  b.addString("Obstacles removed, thank you");
358  Bottle tmp = port_speak_output.prepare();
359  tmp.clear();
360  tmp=b;
361  port_speak_output.write();
362  obstacle_removal_time = yarp::os::Time::now();
363  }
364  }
365  else
366  {
367  if (fabs(current_time-obstacle_time)>max_obstacle_wating_time)
368  {
369  fprintf (stdout, "failed to recover from obstacle, goal aborted \n");
370  status=ABORTED;
371  }
372  }
373  break;
374 
375  case PAUSED:
376  //check if pause is expired
377  double current_time = yarp::os::Time::now();
378  if (current_time - pause_start > pause_duration)
379  {
380  fprintf(stdout, "pause expired! resuming \n");
381  status=MOVING;
382  }
383  break;
384  }
385 
386  if (status != MOVING)
387  {
388  control_out[0]=control_out[1]=control_out[2] = 0.0;
389  }
390 
391  if (enable_retreat && retreat_counter >0)
392  {
393  control_out[0]=180;
394  control_out[1]=0.4;
395  control_out[2]=0;
396  retreat_counter--;
397  }
398 
399  sendOutput();
400  mutex.post();
401 }

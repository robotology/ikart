78  void set_cartesian_laser_data (const yarp::os::Bottle* laser_map)
79  {
80  if (laser_map==0) return;
81  for (unsigned int i=0; i<1080; i++)
82  {
83  Bottle* elem = laser_map->get(i).asList();
84  if (elem ==0) printf ("ERROR\n");
85  laser_x[i] = elem->get(0).asDouble();
86  laser_y[i] = elem->get(1).asDouble();
87  distances[i]=sqrt(laser_x[i]*laser_x[i]+laser_y[i]*laser_y[i]);
88  angles[i] = atan2(double(laser_x[i]),double(laser_y[i]))*RAD2DEG;
89  }
90  }

94  inline const double& get_x (int i) { return laser_x[i]; }
95  inline const double& get_y (int i) { return laser_y[i]; }
96 };

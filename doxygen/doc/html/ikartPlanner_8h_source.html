136  virtual bool threadInit()
137  {
138  //read configuration parametes
139  yarp::os::ResourceFinder mapFinder;
140  mapFinder.setDefaultContext("ikart/maps");
141  mapFinder.configure(0,0);
142 
143  string map_filename;
144  map_filename = mapFinder.getHomeContextPath().c_str() + string("/");
145  map_filename = map_filename + rf.find("map_file").asString().c_str();
146  //map_filename = rf.find("map_file").asString().c_str();
147  if (!map.loadMap(map_filename))
148  {
149  printf("map file not found, closing\n");
150  return false;
151  }
152 
153  if (rf.check("waypoint_tolerance_lin")) {waypoint_tolerance_lin = rf.find("waypoint_tolerance_lin").asDouble();}
154  if (rf.check("waypoint_tolerance_ang")) {waypoint_tolerance_ang = rf.find("waypoint_tolerance_ang").asDouble();}
155  if (rf.check("goal_tolerance_lin")) {goal_tolerance_lin = rf.find("goal_tolerance_lin").asDouble();}
156  if (rf.check("goal_tolerance_ang")) {goal_tolerance_ang = rf.find("goal_tolerance_ang").asDouble();}
157  if (rf.check("use_optimized_path")) {int p=rf.find("use_optimized_path").asInt(); select_optimized_path(p==1);}
158  if (rf.check("max_lin_speed")) {max_lin_speed = rf.find("max_lin_speed").asDouble();}
159  if (rf.check("max_ang_speed")) {max_ang_speed = rf.find("max_ang_speed").asDouble();}
160  if (rf.check("min_lin_speed")) {min_lin_speed = rf.find("min_lin_speed").asDouble();}
161  if (rf.check("min_ang_speed")) {min_ang_speed = rf.find("min_ang_speed").asDouble();}
162  if (rf.check("min_waypoint_distance")) {min_waypoint_distance = rf.find("min_waypoint_distance").asInt();}
163 
164  //open module ports
165  string localName = "/ikartPathPlanner";
166  port_localization_input.open((localName+"/localization:i").c_str());
167  port_laser_map_input.open((localName+"/laser_map:i").c_str());
168  port_status_input.open((localName+"/navigationStatus:i").c_str());
169  port_status_output.open((localName+"/plannerStatus:o").c_str());
170  port_commands_output.open((localName+"/commands:o").c_str());
171  port_map_output.open((localName+"/map:o").c_str());
172  port_yarpview_target_input.open((localName+"/yarpviewTarget:i").c_str());
173  port_yarpview_target_output.open((localName+"/yarpviewTarget:o").c_str());
174 
175  //automatic port connections
176  bool b = false;
177  b = Network::connect("/ikart_ros_bridge/localization:o",(localName+"/localization:i").c_str(), "udp", false);
178  // if (!b) {fprintf (stderr,"Unable to connect the localization port!\n"); return false;}
179  b = Network::connect("/ikart/laser:o",(localName+"/laser:i").c_str(), "udp", false);
180  if (!b) {fprintf (stderr,"Unable to connect the laser port!\n"); }
181  return true;
182  }

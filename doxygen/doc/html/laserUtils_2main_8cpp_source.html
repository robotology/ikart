33 class laserUtilsModule: public RFModule
34 {
35  Port rpcPort;
36  BufferedPort<Vector> port_laser_i1;
37  BufferedPort<Vector> port_laser_i2;
38  BufferedPort<Vector> port_laser_o1;
39  Vector laser1;
40  Vector laser2;
41  Vector laser_out;
42 
43  bool option_merge_lasers;
44 
45 public:
46  laserUtilsModule()
47  {
48  laser1.resize(1080,1000.0);
49  laser2.resize(1080,1000.0);
50  laser_out.resize(1080,1000.0);
51  option_merge_lasers = true;
52  }
53 
54  virtual bool configure(ResourceFinder &rf)
55  {
56  rpcPort.open("/laserUtils/rpc");
57  port_laser_i1.open("/laserUtils/laser1:i");
58  port_laser_i2.open("/laserUtils/laser2:i");
59  port_laser_o1.open("/laserUtils/laser1:o");
60  attach(rpcPort);
61  return true;
62  }
63 
64  bool respond(const Bottle& command, Bottle& reply)
65  {
66  reply.clear();
67  if (command.get(0).asString()=="help")
68  {
69  reply.addString("Available commands are:");
70  reply.addString("no commands");
71  }
72  else
73  {
74  reply.addString("Unknown command.");
75  }
76  return true;
77  }
78 
79  virtual bool close()
80  {
81  rpcPort.close();
82  port_laser_i1.close();
83  port_laser_i2.close();
84  port_laser_o1.close();
85  return true;
86  }
87 
88  virtual double getPeriod()
89  {
90  return 0.01;
91  }
92 
93  virtual bool updateModule()
94  {
95  Vector* v1 = port_laser_i1.read(false);
96  if (v1) laser1 = *v1;
97  Vector* v2 = port_laser_i2.read(false);
98  if (v2) laser2 = *v2;
99  if (option_merge_lasers == true)
100  for (int i=0; i<1080; i++)
101  {
102  if (laser2[i] < laser1[i]) laser_out[i] = laser2[i];
103  else laser_out[i] = laser1[i];
104  }
105  return true;
106  }
107 
108 };

275 void MotorControl::read_inputs(double *linear_speed,double *angular_speed,double *desired_direction, double *pwm_gain)
276 {
277  static double wdt_old_mov_cmd=Time::now();
278  static double wdt_old_joy_cmd=Time::now();
279  static double wdt_old_aux_cmd=Time::now();
280  static double wdt_mov_cmd=Time::now();
281  static double wdt_joy_cmd=Time::now();
282  static double wdt_aux_cmd=Time::now();
283 
284  if (Bottle *b = port_joystick_control.read(false))
285  {
286  if (b->get(0).asInt()==1)
287  {
288  //received a joystick command.
289  read_percent_polar(b, joy_desired_direction,joy_linear_speed,joy_angular_speed,joy_pwm_gain);
290  wdt_old_joy_cmd = wdt_joy_cmd;
291  wdt_joy_cmd = Time::now();
292 
293  //Joystick commands have higher priorty respect to movement commands.
294  //this make the joystick to take control for 100*20 ms
295  if (joy_pwm_gain>10) joystick_received = 100;
296  }
297  }
298  if (Bottle *b = port_movement_control.read(false))
299  {
300  if (b->get(0).asInt()==1)
301  {
302  read_percent_polar(b, cmd_desired_direction,cmd_linear_speed,cmd_angular_speed,cmd_pwm_gain);
303  wdt_old_mov_cmd = wdt_mov_cmd;
304  wdt_mov_cmd = Time::now();
305  command_received = 100;
306  }
307  else if (b->get(0).asInt()==2)
308  {
309  read_speed_polar(b, cmd_desired_direction,cmd_linear_speed,cmd_angular_speed,cmd_pwm_gain);
310  wdt_old_mov_cmd = wdt_mov_cmd;
311  wdt_mov_cmd = Time::now();
312  command_received = 100;
313  }
314  else if (b->get(0).asInt()==3)
315  {
316  read_speed_cart(b, cmd_desired_direction,cmd_linear_speed,cmd_angular_speed,cmd_pwm_gain);
317  wdt_old_mov_cmd = wdt_mov_cmd;
318  wdt_mov_cmd = Time::now();
319  command_received = 100;
320  }
321  }
322  if (Bottle *b = port_auxiliary_control.read(false))
323  {
324  if (b->get(0).asInt()==1)
325  {
326  read_percent_polar(b, aux_desired_direction,aux_linear_speed,aux_angular_speed,aux_pwm_gain);
327  wdt_old_aux_cmd = wdt_aux_cmd;
328  wdt_aux_cmd = Time::now();
329  auxiliary_received = 100;
330  }
331  else if (b->get(0).asInt()==2)
332  {
333  read_speed_polar(b, aux_desired_direction,aux_linear_speed,aux_angular_speed,aux_pwm_gain);
334  wdt_old_aux_cmd = wdt_aux_cmd;
335  wdt_aux_cmd = Time::now();
336  auxiliary_received = 100;
337  }
338  else if (b->get(0).asInt()==3)
339  {
340  read_speed_cart(b, aux_desired_direction,aux_linear_speed,aux_angular_speed,aux_pwm_gain);
341  wdt_old_aux_cmd = wdt_aux_cmd;
342  wdt_aux_cmd = Time::now();
343  auxiliary_received = 100;
344  }
345  }
346 
347  //priority test
348  if (joystick_received>0)
349  {
350  *desired_direction = joy_desired_direction;
351  *linear_speed = joy_linear_speed;
352  *angular_speed = joy_angular_speed;
353  *pwm_gain = joy_pwm_gain;
354  }
355  else if (auxiliary_received>0)
356  {
357  *desired_direction = aux_desired_direction;
358  *linear_speed = aux_linear_speed;
359  *angular_speed = aux_angular_speed;
360  *pwm_gain = aux_pwm_gain;
361  }
362  else //if (command_received>0)
363  {
364  *desired_direction = cmd_desired_direction;
365  *linear_speed = cmd_linear_speed;
366  *angular_speed = cmd_angular_speed;
367  *pwm_gain = cmd_pwm_gain;
368  }
369 
370  //watchdog on received commands
371  static double wdt_old=Time::now();
372  double wdt=Time::now();
373  //fprintf(stderr,"period: %f\n", wdt-wdt_old);
374  if (wdt-wdt_mov_cmd > 0.200)
375  {
376  cmd_desired_direction=0;
377  cmd_linear_speed=0;
378  cmd_angular_speed=0;
379  cmd_pwm_gain=0;
380  mov_timeout_counter++;
381  }
382  if (wdt-wdt_joy_cmd > 0.200)
383  {
384  joy_desired_direction=0;
385  joy_linear_speed=0;
386  joy_angular_speed=0;
387  joy_pwm_gain=0;
388  joy_timeout_counter++;
389  }
390  if (wdt-wdt_aux_cmd > 0.200)
391  {
392  aux_desired_direction=0;
393  aux_linear_speed=0;
394  aux_angular_speed=0;
395  aux_pwm_gain=0;
396  aux_timeout_counter++;
397  }
398 
399  if (wdt-wdt_old > 0.040) { thread_timeout_counter++; }
400  wdt_old=wdt;
401 
402  if (joystick_received>0) { joystick_received--; }
403  if (command_received>0) { command_received--; }
404  if (auxiliary_received>0) { auxiliary_received--; }
405 }

407 void MotorControl::decouple(double appl_linear_speed, double appl_desired_direction, double appl_angular_speed)
408 {
409  static double off = 0;
410  if (wheels_type==WHEEL_TYPE_MECHANUM) off = 45.0; //degrees
411 
412  //wheel contribution calculation
413  FA = appl_linear_speed * cos ((150.0-appl_desired_direction+off)/ 180.0 * 3.14159265) + appl_angular_speed;
414  FB = appl_linear_speed * cos ((030.0-appl_desired_direction+off)/ 180.0 * 3.14159265) + appl_angular_speed;
415  FC = appl_linear_speed * cos ((270.0-appl_desired_direction+off)/ 180.0 * 3.14159265) + appl_angular_speed;
416 }

418 void MotorControl::execute_speed(double appl_linear_speed, double appl_desired_direction, double appl_angular_speed)
419 {
420  decouple(appl_linear_speed,appl_desired_direction,appl_angular_speed);
421 
422  //Use a low pass filter to obtain smooth control
423  if (motors_filter_enabled == 1)
424  {
425  FA = ikart_filters::lp_filter_1Hz(FA,0);
426  FB = ikart_filters::lp_filter_1Hz(FB,1);
427  FC = ikart_filters::lp_filter_1Hz(FC,2);
428  }
429  else if (motors_filter_enabled == 2)
430  {
431  FA = ikart_filters::lp_filter_2Hz(FA,0);
432  FB = ikart_filters::lp_filter_2Hz(FB,1);
433  FC = ikart_filters::lp_filter_2Hz(FC,2);
434  }
435  else if (motors_filter_enabled == 4) //default
436  {
437  FA = ikart_filters::lp_filter_4Hz(FA,0);
438  FB = ikart_filters::lp_filter_4Hz(FB,1);
439  FC = ikart_filters::lp_filter_4Hz(FC,2);
440  }
441  else if (motors_filter_enabled == 8)
442  {
443  FA = ikart_filters::lp_filter_8Hz(FA,0);
444  FB = ikart_filters::lp_filter_8Hz(FB,1);
445  FC = ikart_filters::lp_filter_8Hz(FC,2);
446  }
447 
448  //Apply the commands
449 #ifdef CONTROL_DEBUG
450  fprintf (stdout,">**: %+6.6f %+6.6f **** %+6.6f %+6.6f %+6.6f\n",exec_linear_speed,exec_desired_direction,-FA,-FB,-FC);
451 #endif
452  ivel->velocityMove(0,-FA);
453  ivel->velocityMove(1,-FB);
454  ivel->velocityMove(2,-FC);
455 }
